# koa

koa1基于generator

koa2支持async

HTTP中间件框架

安装模块

npm i -S koa@latest

# 使用koa写一个简单的服务

const koa = require('koa');
const app = new koa();

app.use(async ctx=>{
    ctx.body = "hello world!";
})

app.listen(3000);

下面和原生的nodejs写一个HTTP服务做一个对比：

const http = require('http');

http.createServer((req,res)=>{
    res.write("hello world!");
    res.end();
}).listen(3000);

可以明显看到使用koa后，比原生的要方便许多。

对比node，koa将req和res整合到了一个对象上ctx，且可以直接传数据，而不用手动的使用end方法结束服务

# app.use(function)

将给定的中间件方法添加到此应用程序

# context

koa将res和req分装在ctx对象上

app.use(async (ctx, next) => {
  ctx.assert(ctx.request.accepts('xml'), 406);
  // equivalent to:
  // if (!ctx.request.accepts('xml')) ctx.throw(406);
  await next();
});

app.use(async (ctx, next) => {
  await next();
  ctx.response.type = 'xml';
  ctx.response.body = fs.createReadStream('really_large.xml');
});

# ctx.body====ctx.response.body

设置响应内容，除了字符串和buffer，还可以直接传json对象

# ctx.url===ctx.requset.url

获取请求的URL地址

# ctx.method===ctx.request.method

ctx.req和ctx.request的区别：

  • ctx.request:是Koa2中context经过封装的请求对象，它用起来更直观和简单。
  • ctx.req:是context提供的node.js原生HTTP请求对象。这个虽然不那么直观，但是可以得到更多的内容，适合我们深度编程

获取请求方式

  1. get

    可以使用ctx.query和ctx.querystring解析URL中的查询字符串

ctx.query一般常用这个方法，他可以得到解析后的一个json对象

ctx.querystring只能得到一个字符串

  1. post

    比较麻烦的是处理post请求发送过来的数据

    const koa = require('koa')
const app = new koa();

app.use(async ctx=>{
    console.log(ctx.method);
    if(ctx.method === 'GET' && ctx.url === '/'){
        let html = `<html>
            <head></head>
            <body>
                <form method='post' action='/'>
                <p>userName</p>
                <input name="userName" /><br/>
                <p>age</p>
                <input name="age" /><br/>
                <input type='submit'>
            </body>
        </html>`;
        
      ctx.body = html;
    }else if(ctx.method === 'POST' && ctx.url === '/'){
        let postData = await parsePostData(ctx);
        console.log(postData);
        ctx.body = postData;
    }else{
        ctx.body = `<p>404</p>`
    }
});


function parsePostData(ctx){
    let postData = ""
    return new Promise((res,rej)=>{
        try{
            ctx.req.on('data',(data)=>{
                postData += data;
            });
            console.log(postData);
            ctx.req.on('end',()=>{
                postData = parseQueryStr(postData);
                res(postData);
            })
        }catch(e){
            rej(e);
        }
    })
}

function parseQueryStr(queryStr){
    let queryData={};
    let queryStrList = queryStr.split('&');
    console.log(queryStrList);
    for( let queryStr of queryStrList ){
        let itemList = queryStr.split('=');
        console.log(itemList);
        queryData[itemList[0]] = decodeURIComponent(itemList[1]);
    } 
    return queryData
}
app.listen(3000)

    可以看到使用原生的koa处理非常麻烦，一般我们都使用koa-body或者koa-bodyparse的中间件处理post请求，它们会将请求体中的数据，解析好以后放在ctx.request.body中

# 简易爬虫

安装superagent（ajax请求）、cheerio模块

npm i -S superagent
npm i -S cheerio

const koa = require('koa');
const superagent = require('superagent');
const cheerio = require('cheerio');
const {join} = require('path');
const app = new koa();

app.use(async ctx =>{
    const arr = [];
    const url = "https://www.douyu.com";
    const data = await new Promise( resolve =>{
        superagent
            .get('https://www.douyu.com/g_jdqscjzc')
            .end((err, res) => {
                const data = res.text;
                const $ = cheerio.load(data);
                $("#live-list-contentbox li").each((i, v) => {
                    const $v = $(v);
                    const obj = {
                        name: $v.find("span.dy-name").text(),
                        room: $v.find("a.play-list-link").prop("href").slice(1)
                        // title: $v.find("a.ictxt").text().trim(),
                        // num: Number.parseInt($v.find("a.item-txt").text().slice(5)),
                        // src: join(url, $v.find("a.cimg").prop("href"))
                    }
                    arr.push(obj);
                })
                return resolve(arr);
            });
    })
    ctx.body = data;
})
app.listen(3000,()=>{
    console.log("3000端口启动")
})

# koa-router

koa的路由管理模块

npm i -S koa-router

# koa-router基础使用：

例子：

var Koa = require('koa');
var Router = require('koa-router');

var app = new Koa();
var router = new Router();

router.get('/', (ctx, next) => {
  // ctx.router available
});

app//必须写
  .use(router.routes())
  .use(router.allowedMethods());

# 设置路由前缀

可以是不在项目路径中的

当需要给所有的路径都加一个相同的前缀，可以使用下面这种简单的办法

const router = new Router({
      prefix:'/goudan'
})
//之后的路由都是 localhost:3000/goudan/ 的形式

# 设置路由层级

给不同的路由家不同的层级

const Koa = require('koa');
const app = new Koa();
const Router = require('koa-router');
let home = new Router();
home.get('/goudan',async(ctx)=>{
    ctx.body="Home goudan";
}).get('/todo',async(ctx)=>{
    ctx.body ='Home ToDo';
})
let page = new Router();
page.get('/goudan',async(ctx)=>{
    ctx.body="Page goudan";
}).get('/todo',async(ctx)=>{
    ctx.body ='Page ToDo';
})
//装载所有子路由
let router = new Router();
router.use('/home',home.routes(),home.allowedMethods());
router.use('/page',page.routes(),page.allowedMethods());
//加载路由中间件
app.use(router.routes()).use(router.allowedMethods());
app.listen(3000,()=>{
    console.log('[demo] server is starting at port 3000');
});

# koa使用cookie

  • ctx.cookies.get(name,[optins]):读取上下文请求中的cookie。
  • ctx.cookies.set(name,value,[options])：在上下文中写入cookie。

# Cookie选项

比如我们要存储用户名，保留用户登录状态时，你可以选择7天内不用登录，也可以选择30天内不用登录。这就需要在写入是配置一些选项：

  • domain：写入cookie所在的域名
  • path：写入cookie所在的路径
  • maxAge：Cookie最大有效时长
  • expires：cookie失效时间
  • httpOnly:是否只用http请求中获得
  • overwirte：是否允许重写

# koa-views

该模块可以引入模板引擎

模板引擎有很多，ejs、pug、jade等

const Koa = require('koa')
const views = require('koa-views')
const path = require('path')
const app = new Koa()
/******	加载模板引擎*********/
app.use(views(path.join(__dirname, './view'), {
  extension: 'ejs'//模板文件放在view文件夹下
}))
/*******加载模板引擎*********/
app.use( async ( ctx ) => {
  let title = 'hello koa2';
  /*******渲染模板********/
  await ctx.render('index', {
    title
  })
  /*******渲染模板**********/
})
app.listen(3000,()=>{
    console.log('[demo] server is starting at port 3000');
})
