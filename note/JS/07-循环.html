# 循环

# 1.for循环

for( 语句1 , 语句2 , 语句4 ){
	//语句3 循环体代码
    //code
    //code
}

//执行顺序： 1  ->  2?  -> 3  -> 4 -> 2? -> 3 -> 4 -> 2 ……

# break/continue关键字

break 跳出循环，后面还没执行的循环体代码不执行，同时for结束

continue 跳过当前循环体，后面还没执行的循环体代码不执行，进入到④的迭变跳过当前循环，

switch里面的break只对switch起作用，不会影响外层的for循环

for(i=0;i<10;i++){
    switch(a){
        case 1:
            break;
    }
}

//通过给for循环起名，跳出外层for 循环
aaa:for(i=0;i<10;i++){
    for(j=0;j<10;j++){
        if(j>8){
            break aaa;
        }
    }
}

# while循环

let i=0;
while(i<5){
    console.log(i);
    i++;
}

# do while

先执行循环体代码，再做判断

let x = 800;
do{
    console.log(x);
    x++;
}
while(x<5);

let oWrap = document.getElementById("wrap");

oWrap.onclick = function () {
