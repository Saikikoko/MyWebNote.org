# 数组和ES6新增API

# 修改原数组

  1. # arr.sort()排序

arr.sort(function(a,b){ return a-b;})

  • 默认arr.sort() 以首字符编码大小排序
  • 数组length小于10以冒泡排序
    • 冒泡排序下依次比较 return >0调换位置，=0不调换位置，<0不调换位置
  • 数组length大于10以二分排序

arr.sort(function(a,b){ return a-b;}) 升序排序

arr.sort(function(a,b){ return b-a;}) 降序排序

//常规操作
let a = [12,78,50,20,32,40,90];
a.sort( function(x,y){
    return y-x;
} );

console.log(a);//[90, 78, 50, 40, 32, 20, 12]

//进阶
let a = [
    {name:"阿飞",age:20},
    {name:"朱雀",age:30},
    {name:"丫丫",age:18},
    {name:"心艾",age:80},
    {name:"岚岚",age:22},
    {name:"发发",age:22},
    {name:"艾米",age:28}
];

a.sort(function(x,y){
    return x.age-y.age;
});

console.log(a);

  1. # arr.reverse()反转数组

reverse() 颠倒 数组中元素的顺序 等价于 arr.sort() return始终>0的情况

arr.sort().reverse();//降序排列

# 不改变原数组

  1. # arr.concat()数组拼接

arr.concat(arr1,arr2,…,arrN) :数组拼接 连接两个或更多的数组，（并返回拼接成的新数组）

  • 该数组是通过把所有 arrX参数添加到 arr中生成的。
  • 如果要进行 concat() 操作的参数是数组，那么添加的是数组中的元素，而不是数组

  1. # Array.isArray( ) 判断是不是数组

返回一个布尔值

  1. # **arr.join()拼接成字符串**

let arr = [1,2,3,4,6];
console.log(arr.join("-"));//1-2-3-4-5-6

通常使用.join("")将字符数组转化为字符串

# 遍历

以下下方法都能实现遍历，语法也都一样，只是返回值不一样-------不修改原数组

array.xxxx( function(currentValue,index,arr ) )

  • currentValue当前元素的值
  • index 当期元素的索引值
  • arr当期元素属于的数组对象

  1. # forEach()

arr.forEach() 从头至尾遍历数组------无返回值

/*
                接受函数，该函数第一个形参代表每一位的数据，
                              第二个形参代表每一位的序号，
                              第三个形参代表原数组
            */
let a = [4,5,6];
let b = a.forEach(function(x,y,z){
    console.log(x,y,z);

});

console.log(b);

  1. # map() 返回值数组

arr.map()返回一个数组，包含函数所有返回值

let a = [1,2,4];

let b = a.map( function(x,y,z){
    console.log(x,y,z);

    return x*x /2;
} );

console.log(a);
console.log(b);

  1. # filter()

arr.filter() return 值为true或能转化为true则保留该元素------返回数组

let a = [10,20,5,7,89,41];

let b = a.filter(function(x,y,z){
    return x>=10;
});

console.log(a);
console.log(b);

  1. # every

一直运行直到回调函数返回false或者一个假值

  1. # some(..)

一直运行直到回调函数返回true或者真值

  1. # for of(..)ES6新增方法

var myArray = [1,2,3]

for(var v of myArray){
    console.lpooh(v);
}
//1
//2
//3

它与for in的区别是for in遍历的是数组的下标索引，for of遍历的是数组的值

# ES6新增数组及字符串API

# 1.模板字符串扩展

模板字符串可以用来执行函数：

alert`123`; 
//等同于
alert('123');

fn`a${a+1}b${a+2}c`;
//等同于
fn(["a","b","c"],2,3);

# 2.包含关系的检测

  • includes()：返回布尔值，表示是否找到了参数字符串。
  • startsWith()：返回布尔值，表示参数字符串是否在原字符串的头部。
  • endsWith()：返回布尔值，表示参数字符串是否在原字符串的尾部。

都支持第二个参数，表示检测的起止位置。

let str = "Hello World";

str.includes("Hello"); //true
str.includes("Hello" , 2); //false
str.startsWidth("Hello"); //true
str.startsWidth("World" , 6); //true
str.endsWidth('World'); //true
str.endsWidth('Hello' , 5); //true

# 3.repeat内容重复

返回一个新字符串，表示将原字符串重复n次

"hello".repeat(3); // "hellohellohello"

参数不是正整数数，会进行对应的处理，比如小数的话会取整，负数的话会报错，其他数据类型能转则转等等，但是写代码时一般不会出现这种情况。

# 4.补全

  • padStart() 返回新字符串，前置补全
  • padEnd() 返回新字符串，后置补全

重复第二个参数来补全，太长的截取，没有第二个参数以 空格 补全

"xx".padStart(5,"a"); //"aaaxx"
"xx".padStart(5,"abcdef"); //"abcxx"
"xx".padEnd(5,"abcdef"); //"xxabc"
"xx".padEnd(5); //"x    "

常用与，数位的补全，或者格式的补全

//显示时间的时候，我们做过单个位前面补0的案例
(5+"").padStart(2,"0"); // "05"

//年月日格式
"05-30".padStart(10,"YYYY-MM-DD"); // "YYYY-05-30"

# 5.数组的扩展运算符 ...

...arr 可以将数组转换成用 , 隔开的参数序列：

let arr = [2,3,4];
function f(a,b,c){
    console.log(a);
    console.log(b);
    console.log(c);
}
f(arr);
//[2,3,4]
//undefined
//undefined

f(...arr);
//2
//3
//4

数组拼接

let a = [1,2,3];
let b = [4,5,6];
let c = [...a,...b];// c =  [1,2,3,4,5,6];
b.push(...a) // b = [4,5,6,1,2,3];

类数组变数组

同样的，有Iterator遍历器接口(Array String NodeList arguments 和新增的数据结构Map Set)的都能进行 ... 操作

var aP = document.getElementsByTagName("p");
var arrP = [].slice.call(aP);
	//ES6可以写为
let aP = document.getElementsByTagName("p");
let arrP = [...aP];//arrP是静态的

配合解构一起使用

let [a,...b] = [1,2,3,4]; //a=1  b=[2,3,4]
let [a,...b] = []; //a=undefined  b=[]
