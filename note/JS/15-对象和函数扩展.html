# 对象和函数扩展

# 对象遍历

# for in(..)用来遍历对象的所有可枚举属性名

let a = {
    name : "阿飞",
    age : 18,
    marry : true,
    handsome : true
};

//for in
for( let key in a ){
    console.log( key , ":" , a[key] );
}

# for in也可以遍历数组

但是不推荐使用，因为如果有人对数组添加了可枚举属性如：arr.goudan = "erha"该方法会把该属性也遍历进去

let b= ["狗蛋","老王","小吴"];

for (let key in b ){
    console.log(key , ":" , b[key] );
}
/*
"0": "狗蛋"
"1": "老王"
"2": "小吴"
*/

# json

一个格式非常标准的js对象

  • JSON 指的是 JavaScript 对象表示法（JavaScript Object Notation）
  • JSON 是轻量级的文本数据交换格式
  • JSON 独立于语言：JSON 使用 Javascript语法来描述数据对象，但是 JSON 仍然独立于语言和平台。JSON 解析器和 JSON 库支持许多不同的编程语言。 目前非常多的动态（PHP，JSP，.NET）编程语言都支持JSON。
  • JSON 具有自我描述性，更易理解
let a = {"name":"阿飞","age":18,"marry":true};

JSON.stringify()反序列化

将对象转换成json字符串

let a = {
    'name' : "阿飞",
    age : 18,
    marry : true,
    handsome : true
};

console.log( JSON.stringify(a) );

JSON.parse(b)序列化

将json字符串转换成对象

let b = '{"a":10,"b":"狗蛋","c":true}';

console.log( JSON.parse(b) ); //序列化

函数参数的默认值

//ES5中
function fn(a,b){
    if(b===undefined){
        b = 2;
    }
}

//ES6中
function a(a,b=3){
    alert(1);
}
a(2);//b=3

一般设置函数参数默认值，都会给最后一个参数加上默认值，否则无意义

function fn(a,b=2,c){
    alert(1);
}
fn(2,undefined,4)//此时b传参必须传一个und，否则会报错

# 函数参数的解构赋值

function fn( {x,y} ){
    console.log(x);//10
    console.log(y);//30
}
fn({x:10,y:30});

解构赋值和默认值配合使用

function fn(  {x=10,y=30}  ){
    console.log(x);
    console.log(y);
}
//fn();//报错
fn({}); //10 30

/*******************************/
function fn( {x,y} = {x:10,y:30}  ){
    console.log(x);
    console.log(y);
}
//fn(); //10 30
fn({}); //undefined undefined

# 箭头函数=>

() => {}

//一个形参
let fn = v => v;
/*等价于*/
var fn = function(v){
    return v;
}

//多个形参
var fn = function(a,b){
    return a+b;
}
/*等价于*/

let fn = (a,b) => a+b;

//多条函数与句
var fn = function(a,b){
    let c = a+2;
    let d = b-3;
    alert(c+d);
    return 3;
};
/*等价于*/
let fn =   (a,b)=>{
    let c = a+2;
    let d = b-3;
    alert(c+d);
    return 3;
};

因为{}会被解析成代码块，所以要返回一个对象时，不能直接写{}，而要加()括起来

let f = () => {name:"tom",age:18};//错误的写法

let f = () => （{name:"tom",age:18}）;//正确

注意点

  1. 函数体内的this对象，就是定义时所在的对象，而不是使用时所在的对象。
  2. 不可以当作构造函数，也就是说，不可以使用new命令，否则会抛出一个错误。
  3. 不可以使用arguments对象，该对象在函数体内不存在。如果要用，可以用 rest 参数代替。
  4. 不可以使用yield命令，因此箭头函数不能用作 Generator 函数。

上面四点中，第一点尤其值得注意。this对象的指向是可变的，但是在箭头函数中，它是固定的，它由外层（函数或者全局）作用域来决定。

  1. 箭头函数作为函数的一种形式,对于this的处理和普通函数有所区别,其没有自己的this上下文,也就是说通过bind/call/apply函数方法设置this值时无效的，会被忽略
  2. 因为箭头函数没有自己的this上下文，那么当它作为对象的方法函数运行时，this并不指向这个对象
  3. 箭头函数的的函数体中出现的this在运行时绑定到最近的作用域上下文对象
  4. 你可以认为箭头函数的this和调用者无关，只和其定义时所在的上下文相关
