# 纯函数

相同的输入，永远得到相同的输出，不能改变外界的状态

const a = 1;

function fn(a){
    a = a + 1;
    return a;
}

只需关注函数返回值就行

# 偏函数

固定一个函数的一个或者多个参数，然后返回一个新函数，返回入参函数的剩余函数的接受应用

用来避免一个参数的重复输入

function add(a,b){
    return a + b；
}
//这个工厂接受一个入参函数，返回一个新函数
function partial(fn,a){
    return function(b){
        return fn.call(this,a,b) 
    }
}

const parAdd = partial(add,1);

const obj = {}

obj.parAdd = parAdd

console.log(obj.parAdd(3))

# 柯里化函数

把一个多参(n)函数转化为n个单参函数

目的：为了缩小适用范围，创建一个针对性更强的函数

优点：复用性，延迟计算

function add(a,b){
    return a + b;
}

add(1,2,3,4,5);

const fn = curry(add)

//完整版的柯里化
function curry(fn, minArg){
    if(minArg === undefined) minArg = 1

    function f(fArg){
        return function(){
            var arg = [].slice.call(arguments)
            var newArg = fArg.concat(arg)
            if(newArg.length >= minArg){
                return fn.apply(null, newArg)
            }else{
                return f(newArg)
            }
        }
    }
    return f([])
}

const fn = curry(add, 5)


const f = fn(1)()(3)

console.log(f(4)(5)()(6))

# 节流

为了降低性能损耗，给事件触发设置固定的时间间隔，如鼠标滚轮滚动

事件多次触发，但按照指定时间间隔执行

const fn = function(){
    console.log("scroll")
}
//throttle节流函数
//原理 获取事件的时间戳 若两次事件的时间间隔大于指定值，便执行事件
function throttle(fn,time){
    var timer =null
 	var startTime = new Date()
    
    return function(){
        var args = [].slice.call(arguments)
        var time_ = new Date() - startTime >= time
        if(time_){
            fn.apply(this,args)
            startTime = new Date()
        }
    }
}

const f = throttle(fn,300)

window.onscroll = f

# 防抖

防抖动和节流本质是不一样的。防抖动是将多次执行变为最后一次执行，节流是将多次执行变成每隔一段时间执行。

const fn = function(){
    console.log("scroll")
}
//防抖函数
//原理 两次事件触发的时间间隔小于设定的值，事件还没触发，就被清楚定时器，大于设定的值，则定时器触发
function debounce(fn,time){
    var timer = null
    return function(){
        var that = this
        var args = [].slice.call(arguments)
        
        clearTimeout(timer)
        timer = setTimeout(function(){
            fn.apply(that,args)
        },time);
    }
}

const f = debounce(fn,500)

window.onscroll = f

# 扁平化

数组的扁平化：降维

const arr = [1,[2,,[3,[4],5],6],7]

function flatten(array){
    var res = []
    
    for(let i = 0; i < array.lengh; i++){
        var item = array[i]
        if(Array.isArray(item)){
            res = res.concat(flatten(item))
        }else{
            res.push(item)
        }
    }
    return res
}
//方式二
function flatten2(array){
    return array.reduce(function(prev,item){
        return prev.concat(Array.isArray(item) ? flatten2(item) : item)
    },[])
}
